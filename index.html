Intégration Google Drive API

Synchronisation des données de production avec Google Drive

Configuration de l'API Google Drive

Pour utiliser l'API Google Drive, vous devez d'abord configurer un projet dans Google Cloud Console.

Étapes de configuration:

  1. Allez sur Google Cloud Console
  2. Créez un nouveau projet ou sélectionnez un projet existant
  3. Activez l'API Google Drive
  4. Créez des identifiants OAuth 2.0
  5. Configurez l'écran de consentement OAuth
  6. Ajoutez votre domaine aux autorisations (localhost pour le développement)

Intégration avec votre application

Voici comment intégrer l'API Google Drive à votre application de suivi de production:

// Chargement de l'API Google
function loadGAPI() {
  gapi.load('client:auth2', initClient);
}

// Initialisation du client Google
function initClient() {
  gapi.client.init({
    apiKey: 'VOTRE_CLE_API',
    clientId: 'VOTRE_ID_CLIENT',
    discoveryDocs: ["https://www.googleapis.com/discovery/v1/apis/drive/v3/rest"],
    scope: "https://www.googleapis.com/auth/drive.file"
  }).then(function () {
    // Gestion de l'authentification
    gapi.auth2.getAuthInstance().isSignedIn.listen(updateSigninStatus);
    updateSigninStatus(gapi.auth2.getAuthInstance().isSignedIn.get());
  });
}

// Connexion à Google Drive
function signIn() {
  gapi.auth2.getAuthInstance().signIn();
}

// Déconnexion de Google Drive
function signOut() {
  gapi.auth2.getAuthInstance().signOut();
}

// Sauvegarder les données sur Google Drive
function saveToDrive(data, filename) {
  const fileContent = JSON.stringify(data);
  const file = new Blob([fileContent], {type: "application/json"});
  const metadata = {
    name: filename,
    mimeType: "application/json"
  };

  const accessToken = gapi.auth.getToken().access_token;
  const form = new FormData();
  form.append('metadata', new Blob([JSON.stringify(metadata)], {type: 'application/json'}));
  form.append('file', file);

  fetch('https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart', {
    method: 'POST',
    headers: new Headers({'Authorization': 'Bearer ' + accessToken}),
    body: form
  }).then((response) => response.json())
    .then((data) => console.log('Fichier sauvegardé:', data))
    .catch((error) => console.error('Erreur:', error));
}

// Charger les données depuis Google Drive
function loadFromDrive(fileId) {
  const accessToken = gapi.auth.getToken().access_token;
  fetch(`https://www.googleapis.com/drive/v3/files/${fileId}?alt=media`, {
    headers: new Headers({'Authorization': 'Bearer ' + accessToken})
  }).then((response) => response.json())
    .then((data) => {
      console.log('Données chargées:', data);
      // Mettre à jour votre application avec ces données
    })
    .catch((error) => console.error('Erreur:', error));
}

Fonctionnement de la synchronisation

Une fois l'API configurée, voici comment synchroniser vos données:

  1. L'utilisateur se connecte à son compte Google via votre application
  2. Votre application sauvegarde les données dans un fichier JSON sur Google Drive
  3. Sur un autre poste, l'utilisateur se connecte au même compte Google
  4. Votre application charge les données depuis Google Drive
  5. Les données sont synchronisées entre tous les postes
La synchronisation avec Google Drive permet un accès multi-postes sans transfert manuel de fichiers.

Alternative: Google Picker API

Pour une expérience utilisateur plus simple, vous pouvez utiliser Google Picker API:

// Charger Google Picker API
function loadPicker() {
  gapi.load('picker', initPicker);
}

// Initialiser Google Picker
function initPicker() {
  const picker = new google.picker.PickerBuilder()
    .addView(google.picker.ViewId.DOCS)
    .setOAuthToken(gapi.auth.getToken().access_token)
    .setDeveloperKey('VOTRE_CLE_DEVELOPPEUR')
    .setCallback(pickerCallback)
    .build();
  picker.setVisible(true);
}

// Callback pour Google Picker
function pickerCallback(data) {
  if (data.action === google.picker.Action.PICKED) {
    const fileId = data.docs[0].id;
    loadFromDrive(fileId);
  }
}
